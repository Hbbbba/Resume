% pdf/header.tex
% Style target: simple academic CV like the provided sample PDF (all-caps section titles, compact).

% ----- Page layout (no geometry) -----
\setlength{\oddsidemargin}{-0.10in}
\setlength{\evensidemargin}{-0.10in}
\setlength{\textwidth}{6.7in}
\setlength{\topmargin}{-0.55in}
\setlength{\textheight}{9.6in}

% ----- Paragraph spacing -----
\setlength{\parindent}{0pt}
\setlength{\parskip}{2.5pt}

% ----- Make section titles look like "EDUCATION / PUBLICATION ..." -----
\makeatletter
% Remove numbering
\renewcommand\thesection{}
\renewcommand\thesubsection{}

% Compact, bold, ALL CAPS section title
\renewcommand\section{\@startsection{section}{1}{0pt}%
  {-0.9ex plus -0.2ex minus -0.2ex}%
  {0.35ex plus 0.1ex}%
  {\normalfont\bfseries\MakeUppercase}}

% Subsection: bold, normal case (used rarely in CV)
\renewcommand\subsection{\@startsection{subsection}{2}{0pt}%
  {-0.7ex plus -0.2ex minus -0.2ex}%
  {0.25ex plus 0.1ex}%
  {\normalfont\bfseries}}
\makeatother

% ----- Compact lists without enumitem -----
\makeatletter
\let\olditemize\itemize
\let\endolditemize\enditemize
\renewenvironment{itemize}{
  \olditemize
  \setlength{\itemsep}{1pt}
  \setlength{\parsep}{0pt}
  \setlength{\parskip}{0pt}
  \setlength{\topsep}{2pt}
}{
  \endolditemize
}

\let\oldenumerate\enumerate
\let\endoldenumerate\endenumerate
\renewenvironment{enumerate}{
  \oldenumerate
  \setlength{\itemsep}{1pt}
  \setlength{\parsep}{0pt}
  \setlength{\parskip}{0pt}
  \setlength{\topsep}{2pt}
}{
  \endoldenumerate
}
\makeatother

% ----- Make links clean (Pandoc often injects hyperref; this won't error if absent) -----
\AtBeginDocument{%
  \providecommand{\hypersetup}[1]{}%
  \hypersetup{hidelinks}%
}

% ----- Keep layout stable -----
\sloppy
\widowpenalty=10000
\clubpenalty=10000
